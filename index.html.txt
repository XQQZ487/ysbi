let current = 0;
let answers = Array(24).fill(null);

const $ = id => document.getElementById(id);
const show = (id) => { $(id).classList.remove('hidden'); };
const hide = (id) => { $(id).classList.add('hidden'); };

// 加载题目
function load() {
    const q = questions[current];
    $('question-title').textContent = `${current+1}. ${q.q}`;
    $('progress-text').textContent = `${current+1}/24`;
    $('progress-bar').style.width = ((current+1)/24*100) + '%';
    
    const opt = $('options-container');
    opt.innerHTML = "";
    const list = ["A. 非常符合我","B. 比较符合我","C. 不太符合我","D. 完全不符合我"];
    const score = [1,2,3,4];
    
    list.forEach((t,i)=>{
        const btn = document.createElement('div');
        btn.className = "p-3 border rounded-lg cursor-pointer hover:border-primary hover:bg-primary/5 transition-all";
        btn.textContent = t;
        if(answers[current] === score[i]) btn.classList.add('border-primary','bg-primary/5');
        btn.onclick = () => {
            answers[current] = score[i];
            load();
        };
        opt.appendChild(btn);
    });

    $('prev-btn').disabled = current === 0;
    $('prev-btn').classList.toggle('opacity-50', current===0);
    $('prev-btn').classList.toggle('cursor-not-allowed', current===0);
    $('next-btn').textContent = current ===23 ? "查看结果" : "下一题 →";
}

// 维度题号严格对应 各6题
function calc() {
    const s = answers;
    // 作息6题：1/6/10/14/19/24
    const 作息 = s[0] + s[5] + s[9] + s[13] + s[18] + s[23];
    // 体态6题：3/7/11/15/20/23
    const 体态 = s[2] + s[6] + s[10] + s[14] + s[19] + s[22];
    // 饮食6题：2/5/9/13/18/22
    const 饮食 = s[1] + s[4] + s[8] + s[12] + s[17] + s[21];
    // 精神6题：4/8/12/16/17/21
    const 精神 = s[3] + s[7] + s[11] + s[15] + s[16] + s[20];

    // 判定线改为 17分
    // ＞17 = 健康(M/P/U/W)
    // ≤17 = 消极脆皮(N/Q/V/X)
    const M = 作息 > 17;
    const P = 体态 > 17;
    const U = 饮食 > 17;
    const W = 精神 > 17;

    const code = (M?"M":"N") + (P?"P":"Q") + (U?"U":"V") + (W?"W":"X");
    const res = personality[code] || "未知人格";
    
    $('score1').textContent = 作息;
    $('score2').textContent = 体态;
    $('score3').textContent = 饮食;
    $('score4').textContent = 精神;
    $('type-code').textContent = code;
    $('type-name').textContent = res.split("\n")[0];
    $('result-desc').innerHTML = res.replace(/\n/g,'<br>');
}
